CREATE DATABASE electrondb;

USE electrondb;

CREATE TABLE product(
  id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
marca VARCHAR(100) NOT NULL,
modelo VARCHAR(100) NOT NULL,
serialFabricante VARCHAR(100) NOT NULL,
activoFijo VARCHAR(100) NOT NULL,
proveedor VARCHAR(100) NOT NULL,
fechaCompra VARCHAR(100) NOT NULL,
estatus VARCHAR(100) NOT NULL,
motivo VARCHAR(100) NOT NULL,
origen VARCHAR(100) NOT NULL,
envio VARCHAR(100) NOT NULL,
ref VARCHAR(100) NOT NULL,
ubicacionFisica VARCHAR(100) NOT NULL,
observacion VARCHAR(100) NOT NULL,
fechaGrafico VARCHAR(100) NOT NULL,
concatenarMF VARCHAR(100) NOT NULL,
cantidad VARCHAR(100) NOT NULL,
);
